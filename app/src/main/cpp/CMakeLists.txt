cmake_minimum_required(VERSION 3.22.1)

project("nodebotdashboard")

add_library(native-lib SHARED native-lib.cpp)

find_library(log-lib log)

# Include Node.js headers
include_directories(${CMAKE_SOURCE_DIR}/../../../libnode/include/node)

# Declare imported Node library
add_library(node SHARED IMPORTED)

set_target_properties(node PROPERTIES IMPORTED_LOCATION
        ${CMAKE_SOURCE_DIR}/../../../libnode/bin/${ANDROID_ABI}/libnode.so)

# Link libraries
target_link_libraries(native-lib node ${log-lib})
